func newPricingCreateCmd() *cobra.Command {
    var (name string; dimType string; price float64)
    c := &cobra.Command{
        Use: "create <model-id>", Args: cobra.ExactArgs(1),
        RunE: func(c *cobra.Command, args []string) error {
            modelID := args[0]
            path := fmt.Sprintf("/v2/api/sources/ai/models/%s/pricing/dimensions", modelID)
            body := map[string]interface{}{"name": name, "unitPrice": price}
            var result map[string]interface{}
            if err := cmd.APIClient.Do(c.Context(), "POST", path, body, &result); err != nil { return err }
            return renderPricingDimension(result)
        },
    }
    c.Flags().StringVar(&name, "name", "", "Dimension name")
    _ = c.MarkFlagRequired("name")
    return c
}

func newPricingUpdateCmd() *cobra.Command {
    var (name string; dimType string; price float64)
    c := &cobra.Command{
        Use: "update <model-id> <dimension-id>", Args: cobra.ExactArgs(2),
        RunE: func(c *cobra.Command, args []string) error {
            body := make(map[string]interface{})
            if c.Flags().Changed("name") { body["name"] = name }
            if len(body) == 0 { return fmt.Errorf("no fields specified to update") }
            path := fmt.Sprintf("/v2/api/sources/ai/models/%s/pricing/dimensions/%s", args[0], args[1])
            var result map[string]interface{}
            if err := cmd.APIClient.Do(c.Context(), "PUT", path, body, &result); err != nil { return err }
            return renderPricingDimension(result)
        },
    }
    return c
}
```

func newPricingDeleteCmd() *cobra.Command {
    c := &cobra.Command{
        Use: "delete <model-id> <dimension-id>", Args: cobra.ExactArgs(2),
        RunE: func(c *cobra.Command, args []string) error {
            dimID := args[1]
            yes, _ := c.Flags().GetBool("yes")
            ok, err := resource.ConfirmDelete("pricing dimension", dimID, yes, cmd.Output.IsJSON())
            if err != nil { return err }
            if !ok { return nil }
            path := fmt.Sprintf("/v2/api/sources/ai/models/%s/pricing/dimensions/%s", args[0], dimID)
            if err := cmd.APIClient.Do(c.Context(), "DELETE", path, nil, nil); err != nil { return err }
            if !cmd.Output.IsQuiet() { fmt.Fprintf(c.OutOrStdout(), "Deleted pricing dimension %s.\n", dimID) }
            return nil
        },
    }
    return c
}
```

API endpoints for alerts:
- List alerts:        GET /v2/api/sources/ai/alert
- Get alert:          GET /v2/api/sources/ai/alert/{id}
- Create alert rule:  POST /v2/api/sources/ai/anomaly (creates anomaly that generates alerts)
- Budget portfolio:   GET /v2/api/ai/alerts/budgets/portfolio
- Budget progress:    GET /v2/api/ai/alerts/{anomalyId}/budget/progress

Budget CRUD proxies to anomaly endpoints (CUMULATIVE_USAGE type):
- Budget create:  POST   /v2/api/sources/ai/anomaly (with type=CUMULATIVE_USAGE + budget fields)
- Budget update:  PUT    /v2/api/sources/ai/anomaly/{id}
- Budget delete:  DELETE /v2/api/sources/ai/anomaly/{id}

Budget progress response fields:
  currentValue, remainingBudget, percentUsed, aheadBehind, budgetThreshold, currency
