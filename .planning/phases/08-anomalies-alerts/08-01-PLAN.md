```go
func newDeleteCmd() *cobra.Command {
    c := &cobra.Command{
        Use: "delete <id>", Args: cobra.ExactArgs(1),
        RunE: func(c *cobra.Command, args []string) error {
            id := args[0]
            yes, _ := c.Flags().GetBool("yes")
            ok, err := resource.ConfirmDelete("product", id, yes, cmd.Output.IsJSON())
            if err != nil { return err }
            if !ok { return nil }
            if err := cmd.APIClient.Do(c.Context(), "DELETE", path+id, nil, nil); err != nil { return err }
            if !cmd.Output.IsQuiet() { fmt.Fprintf(c.OutOrStdout(), "Deleted product %s.\n", id) }
            return nil
        },
    }
    return c
}
```
