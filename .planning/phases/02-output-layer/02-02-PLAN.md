    2. Update the error handling block in `main()`:
       ```go
       func main() {
           if err := cmd.Execute(); err != nil {
               if cmd.JSONMode() {
                   // In JSON mode, errors go to stderr as JSON
                   if apiErr, ok := err.(*apierrors.APIError); ok {
                       f := output.New(true, false)
                       f.RenderJSONError(apiErr.Message, apiErr.StatusCode)
                   } else {
                       f := output.New(true, false)
                       f.RenderJSONError(err.Error(), 0)
                   }
               } else {
                   fmt.Fprintln(os.Stderr, apierrors.RenderError(err.Error()))
               }
               os.Exit(1)
           }
       }
       ```
    3. The key behavior: when --json is active and a command fails, the error is rendered as `{"error": "...", "status": N}` on stderr instead of the styled Lip Gloss error box. Exit code is still non-zero.
